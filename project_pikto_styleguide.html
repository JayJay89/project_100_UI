Shadows

Constructor

@mixin box-shadow($top, $left, $blur, $color, $inset: false) { @if $inset { -webkit-box-shadow:inset $top $left $blur $color; -moz-box-shadow:inset $top $left $blur $color; box-shadow:inset $top $left $blur $color; } @else { -webkit-box-shadow: $top $left $blur $color; -moz-box-shadow: $top $left $blur $color; box-shadow: $top $left $blur $color; } }

Applying As Classes

.inset-shadow { @include box-shadow (0px 2px 1px 0px rgba(black, 0.2) inset); } .outer-shadow { @include box-shadow (1px 1px 4px 0px rgba(black ,0.2)); } .no-shadow { @include box-shadow (none); }
